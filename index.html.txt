        const canvas = document.getElementById('canvas');
        const container = document.getElementById('container');
        const svg = document.getElementById('svg-connections');
        const nodesContainer = document.getElementById('nodes-container');

        // Initialization
        function init() {
            renderNodes();
            renderLines();
            renderRefs();
            updateTransform();
            lucide.createIcons();
        }

        function renderNodes() {
            nodesContainer.innerHTML = nodes.map(node => `
                <div class="node-card" id="node-${node.id}" 
                     style="left: ${node.x}px; top: ${node.y}px; width: ${node.width}px;"
                     onclick="handleNodeClick('${node.id}')">
                    <div class="rounded-2xl shadow-xl border-2 overflow-hidden bg-white ${node.borderColor}">
                        <div class="p-4 flex items-center gap-3 ${node.color} ${node.textColor}">
                            <i data-lucide="${node.icon}" class="${node.id === 'center' ? 'w-6 h-6' : 'w-5 h-5'}"></i>

                            <div class="p-4 bg-white">
                                <p class="text-sm text-slate-700 text-center leading-relaxed font-medium">

                                </p>
                            </div>
                        ` : `

                                Ver Concepto, Importancia y Ejemplo
                            </div>
                        `}
                    </div>
                </div>
            `).join('');
        }

        function renderLines() {
            svg.innerHTML = nodes.filter(n => n.id !== 'center').map(node => `
                <line id="line-${node.id}" class="line" 
                      x1="${center.x}" y1="${center.y}" 
                      x2="${node.x}" y2="${node.y}" />
            `).join('');
        }

        function renderRefs() {
            const list = document.getElementById('ref-list');
            list.innerHTML = references.map(ref => `
                <div class="pl-6 -indent-6 text-sm text-slate-700 leading-relaxed border-b border-slate-50 pb-3 last:border-0">
                    ${ref}
                </div>
            `).join('');
        }

        function handleNodeClick(id) {
            const node = nodes.find(n => n.id === id);
            if (!node || node.id === 'center') return;

            // Update panel content
            document.getElementById('panel-title').innerText = node.title;
            document.getElementById('panel-content').innerText = node.content;
            document.getElementById('panel-importance').innerText = node.importance;
            document.getElementById('panel-example').innerText = node.example;
            document.getElementById('panel-header').className = `p-6 ${node.color} text-white flex justify-between items-start`;
            
            // Highlight Line
            document.querySelectorAll('.line').forEach(l => l.classList.remove('active'));
            document.getElementById(`line-${id}`).classList.add('active');

            // Open Panel
            document.getElementById('side-panel').classList.add('open');
            lucide.createIcons();
        }

        function closePanel() {
            document.getElementById('side-panel').classList.remove('open');
            document.querySelectorAll('.line').forEach(l => l.classList.remove('active'));
        }

        function toggleRefs(show) {
            document.getElementById('ref-modal').classList.toggle('open', show);
        }

        // Pan & Zoom Logic
        container.addEventListener('mousedown', (e) => {
            if (e.target.closest('.node-card')) return;
            isDragging = true;
            dragStart = { x: e.clientX - pos.x, y: e.clientY - pos.y };
        });

        window.addEventListener('mousemove', (e) => {
            if (!isDragging) return;
            pos.x = e.clientX - dragStart.x;
            pos.y = e.clientY - dragStart.y;
            updateTransform();
        });

        window.addEventListener('mouseup', () => isDragging = false);

        function adjustZoom(delta) {
            zoom = Math.max(0.4, Math.min(1.5, zoom + delta));
            updateTransform();
        }

        function resetView() {
            zoom = 0.8;
            pos = { x: 0, y: 0 };
            closePanel();
            updateTransform();
        }

        function updateTransform() {
            canvas.style.transform = `translate(${pos.x}px, ${pos.y}px) scale(${zoom})`;
        }

        // Init on load
        window.onload = init;
